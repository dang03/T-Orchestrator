def parse_json(message)
	# Check JSON message format
	begin
		parsed_message = JSON.parse(message) # parse json message
	rescue JSON::ParserError => e
		# If JSON not valid, return with errors
		logger.error "JSON parsing: #{e.to_s}"
		return message, e.to_s + "\n"
	end

	return parsed_message, nil
end

post '/ns-instances' do
     
     # Return if content-type is invalid
     return 415 unless request.content_type == 'application/json'
     
     # Validate JSON format
     ns, errors = parse_json(request.body.read)
  return 400, errors.to_json if errors

     #check if the request NS_id exist
     begin
         RestClient.get settings.ns_catalogue + '/network-services/' + ns['ns-id']
  rescue => e
		logger.error e
		halt e.response.code, e.response.body
     end
     
     #make a request to the NS Mapping
     #RestClient.post settings.ns_mapping + '/ns/', ns.to_json, :content_type => :json
     
     #Send instantiation to NS Instance repository
     instance = { :ns_id => ns['ns-id'], :status => "instantiated"}
     RestClient.post settings.ns_instance_repository + '/ns-instances', instance.to_json, :content_type => :json
     
     #Send instantiation to the VNF Manager
     
     return 200
end
