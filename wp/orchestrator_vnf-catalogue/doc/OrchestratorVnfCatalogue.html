def create_vnf_object(vnf)
	# Create VNF object
	obj = Vnf.new(name: vnf['name'], vnf_image: vnf['vnf-image'], vnf_manager: vnf['vnf-manager'])

	# Create VNFD object
	vnfd = vnf['vnfd']
	obj.vnfd = Vnfd.new(vendor: vnfd['vendor'], descriptor_version: vnfd['descriptor_version'], version: vnfd['version'], auto_scale_policy: vnfd['auto_scale_policy'], manifest_file: vnfd['manifest_file'], manifest_file_security: vnfd['manifest_file_security'])

	# Create Deployment Flavour object
	deployment_flavour = vnfd['deployment_flavour']
	obj.vnfd.deployment_flavour = DeploymentFlavour.new(flavour_key: deployment_flavour['flavour_key'])

	# Create Constituent VDU objects
	constituent_vdus = deployment_flavour['constituent_vdu']
	constituent_vdus.each do |constituent_vdu|
		obj.vnfd.deployment_flavour.constituent_vdus.push(ConstituentVdu.new(vdu_reference: constituent_vdu['vdu_reference'], number_of_instances: constituent_vdu['number_of_instances'], constituent_vnfc: constituent_vdu['constituent_vnfc']))
	end

	# Create VDU objects
	vdus = vnfd['vdus']
	vdus.each do |vdu|
		vdu_obj = Vdu.new(vm_image: vdu['vm_image'], computation_requirement: vdu['computation_requirement'], virtual_memory_resource_element: vdu['virtual_memory_resource_element'], lifecycle_events: vdu['lifecycle_events'], high_availability: vdu['high_availability'], scale_in_out: vdu['scale_in_out'], openstack_flavour: vdu['openstack_flavour'], cpu_instruction_set_extension: vdu['cpu']['cpu_instruction_set_extension'], cpu_model: vdu['cpu']['cpu_model'], cpu_core_reservation: vdu['cpu']['cpu_core_reservation'], number_of_large_pages_required_per_vdu: vdu['memory']['number_of_large_pages_required_per_vdu'], hypervisor_type: vdu['hypervisor']['hypervisor_type'], hypervisor_version: vdu['hypervisor']['hypervisor_version'], hypervisor_address_translation_support: vdu['hypervisor']['hypervisor_address_translation_support'])

		# Create VNFC objects
		vnfcs = vdu['vnfc']
		vnfcs.each do |vnfc|
			vnfc_obj = Vnfc.new

			# Create Connection Point objects
			connection_points = vnfc['connection_point']
			connection_points.each do |connection_point|
				vnfc_obj.connection_points.push(ConnectionPoint.new(virtual_link_reference: connection_point['virtual_link_reference'], virtual_network_bandwidth_resource_unit: connection_point['virtual_network_bandwidth_resource_unit'], type: connection_point['type']))
			end

			vdu_obj.vnfcs.push(vnfc_obj)
		end

		obj.vnfd.vdus.push(vdu_obj)
	end

	logger.debug obj
	obj
end

def parse_json(message)
	# Check JSON message format
	begin
		parsed_message = JSON.parse(message) # parse json message
	rescue JSON::ParserError => e
		# If JSON not valid, return with errors
		logger.error "JSON parsing: #{e.to_s}"
		return message, e.to_s + "\n"
	end

	return parsed_message, nil
end

post '/vnfs' do
	# Return if content-type is invalid
	return 415 unless request.content_type == 'application/json'

	# Validate JSON format
	vnf, errors = parse_json(request.body.read)
	return 400, errors.to_json if errors

	# Validate VNFD
	begin
		RestClient.post settings.vnfd_validator + '/vnfds', vnf['vnfd'].to_json, :content_type => :json, :accept => :json
	rescue => e
		logger.error e.response
		return e.response.code, e.response.body
	end

	# Create an ActiveRecord object from the VNF in JSON
	vnf = create_vnf_object(vnf)

	# Save to BD
	if vnf.save
		logger.debug vnf
	else
		return 400, vnf.errors.messages
	end

	return 200, {id: vnf.id, name: vnf.name, 'vnf-image' => vnf.vnf_image, created_at: vnf.created_at, updated_at: vnf.updated_at}.to_json
end
