def create_ns_object(ns)
	cp = ConnectionPoint.create(ns['nsd']['connection_points'])
	mp = MonitoringParameter.create(ns['nsd']['monitoring_parameters'])
	#ServiceDeploymentFlavour
	sdfs = []
	cvs = []
	s = ns['nsd']['service_deployment_flavour']
	s.each {|x|
		#service deployment iteration
		x['constituent_vnf'].each {|y|
			cvs << ConstituentVnf.create(y)
		}
		
		x['constituent_vnf'] = cvs
		sdfs <<  ServiceDeploymentFlavour.create(x)
	}
	#SlaSpecification
	sss = []
	aps = []
	s = ns['nsd']['sla_specification']
	s.each {|x|
		#sla_specification iteration
		x['assurance_parameters'].each {|y|
			print y.to_json
			aps << AssuranceParameter.create(y)
		}
		
		x['assurance_parameters'] = aps
		sss <<  SlaSpecification.create(x)
	}
	
	ns['nsd']['service_deployment_flavour'] = sdfs
	ns['nsd']['sla_specification'] = sss
	ns['nsd']['connection_points'] = cp
	ns['nsd']['monitoring_parameters'] = mp
	new_ns = Ns.new(ns['nsd'])
	return new_ns
end

def parse_json(message)
	# Check JSON message format
	begin
		parsed_message = JSON.parse(message) # parse json message
	rescue JSON::ParserError => e
		# If JSON not valid, return with errors
		logger.error "JSON parsing: #{e.to_s}"
		return message, e.to_s + "\n"
	end

	return parsed_message, nil
end
